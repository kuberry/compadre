558  \param p [in] - parameter to be given to the kernel (see Wab definition for context).
559  \param n [in] - parameter to be given to the kernel (see Wab definition for context).
560  */
561  KOKKOS_INLINE_FUNCTION
562  static double Wab(const double r, const double h, const WeightingFunctionType& weighting_type, const int p, const int n) {
563  if (weighting_type == WeightingFunctionType::Power) {
564  // compactly supported on [0,h]
565  // (1 - |r/h|^n)^p
566  // p=0,n=1 -> Uniform, boxcar
567  // p=1,n=1 -> triangular
568  // p=0,n=2 -> Epanechnikov, parabolic
569  // p=2,n=2 -> Quartic, biweight
570  // p=3,n=2 -> Triweight
571  // p=3,n=3 -> Tricube
572  double abs_r_over_h_to_n = std::abs(r/h);
573  if (n>1) abs_r_over_h_to_n = std::pow(abs_r_over_h_to_n, n);
574  return std::pow(1.0-abs_r_over_h_to_n, p) * double(1.0-abs_r_over_h_to_n>0.0);
575  } else if (weighting_type == WeightingFunctionType::CubicSpline) {
576  // compactly supported on [0,h]
577  // invariant to p and n
578  double x = std::abs(r/h);
579  return ((1-x)+x*(1-x)*(1-2*x)) * double(x<=1.0);
580  } else if (weighting_type == WeightingFunctionType::Cosine) {
581  // compactly supported on [0,h]
582  double pi = 3.14159265358979323846;
583  double abs_r_over_h_to_n = std::abs(r/h);
584  return std::cos(0.5*pi*r/h) * double(1.0-abs_r_over_h_to_n>0.0);
585  } else if (weighting_type == WeightingFunctionType::Gaussian) {
586  // NOT compactly supported on [0,h], but approximately 0 at h with >> p
587  // invariant to n, p is number of standard deviations at distance h
588  // 2.5066282746310002416124 = sqrt(2*pi)
589  double h_over_p = h/p;
590  double abs_r_over_h_to_n = std::abs(r/h);
591  return double(1.0-abs_r_over_h_to_n>0.0)/( h_over_p * 2.5066282746310002416124 ) * std::exp(-.5*r*r/(h_over_p*h_over_p));
592  } else if (weighting_type == WeightingFunctionType::Sigmoid) {
593  // NOT compactly supported on [0,h], but approximately 0 at h with >> p
594  // n=0 is sigmoid, n==2 is logistic, with larger p making Wab decay more quickly
595  double abs_r_over_h_to_n = std::abs(r/h);
596  return double(1.0-abs_r_over_h_to_n>0.0) / (std::exp(p*r) + std::exp(-p*r) + n);
597  } else { // unsupported type
598  compadre_kernel_assert_release(false && "Invalid WeightingFunctionType selected.");
599  return 0;
600  }
601  }
