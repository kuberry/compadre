74  //! Space in which to reconstruct polynomial
75  enum ReconstructionSpace {
76  //! Scalar polynomial basis centered at the target site and scaled by sum of basis powers
77  //! e.g. \f$(x-x_t)^2*(y-y_t)*(z-z_t)^3/factorial(2+1+3)\f$ would be a member of 3rd order in 3D, where
78  //! \f$(x_t,y_t,z_t)\f$ is the coordinate of the target site in 3D coordinates.
79  ScalarTaylorPolynomial,
80  //! Vector polynomial basis having # of components _dimensions, or (_dimensions-1) in the case of manifolds)
81  VectorTaylorPolynomial,
82  //! Scalar basis reused as many times as there are components in the vector
83  //! resulting in a much cheaper polynomial reconstruction
84  VectorOfScalarClonesTaylorPolynomial,
85  //! Divergence-free vector polynomial basis
86  DivergenceFreeVectorTaylorPolynomial,
87  };
